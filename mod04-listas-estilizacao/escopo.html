#2: Estilizando inline dinamicamente

"use client"

import { useState } from "react";

const App = () => {

  const [clicado, setClicado] = useState(false);

  const handleClick = () => {
    setClicado(!clicado);
  }

  return (
    <div className="w-screen h-screen flex justify-center items-center">

      <button onClick={handleClick} style={{
        backgroundColor: !clicado ? '#f00' : '#000',
        color: '#fff',
        padding: '3px 15px',
        borderRadius: '10px'
      }}>Clique aqui</button>

    </div>
  );

}

export default App;

#9: Responsividade com Styled

---------

---------
import styled from "styled-components";


type ContainerProps = {
    cor?: string;
} 

export const Container = styled.div`
    max-width: 500px;
    margin: 0 auto;
    background-color: ${props => props.cor};
    color: white;
    padding: 20px;

    span {
        font-weight: bold;
        color: #000;
    }

    .link {
        color: #fff;

        &:hover{ 
            color: #f00;
        }
    }
    
    @media(max-width: 600px){
        display: flex;
        flex-direction:column;
    

    }
`;

type BotaoProps = {
    bg: string;
    small?: boolean;
} 

export const Botao = styled.button`
    font-size: ${props => props.small ? '15px' : '30px'};
    padding: 3px 15px;
    margin: 0 10px;
    background: ${props => props.bg};
    border-radius: 15px;
`;
